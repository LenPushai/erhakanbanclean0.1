// patch-line-items.cjs
// Run with: node patch-line-items.cjs
// This does exactly what your handover prompt requires:
// 1. Full atomic replacement of the lineItems block (no adjacent JSX error)
// 2. Adds all new state + handlers after attachments state
// 3. Fixes uom mapping in handleSaveOrder
// 4. Handles Windows CRLF correctly
// 5. Creates backup first

const fs = require('fs');
const path = require('path');

const FILE_PATH = path.join(__dirname, 'src', 'App.tsx');
const BACKUP_PATH = FILE_PATH + '.bak12'; // new safe backup

console.log('ERHA KANBAN PATCH — Starting...');

// 1. Create backup
if (fs.existsSync(FILE_PATH)) {
  fs.copyFileSync(FILE_PATH, BACKUP_PATH);
  console.log('✅ Backup created → src/App.tsx.bak12');
} else {
  console.error('❌ src/App.tsx not found!');
  process.exit(1);
}

// 2. Read file + normalize to LF
let content = fs.readFileSync(FILE_PATH, 'utf8');
content = content.replace(/\r\n/g, '\n'); // normalize CRLF

// 3. ADD STATE + HANDLERS (right after attachments state)
const attachmentsMarker = 'const [attachments, setAttachments] = React.useState';
const stateToInsert = `
const [newLineItems, setNewLineItems] = React.useState<{
  description: string; quantity: number; uom: string; item_type: string
}[]>([]);
const [savingLines, setSavingLines] = React.useState(false);

const addNewLine = () => setNewLineItems(p => [...p, {description:'',quantity:1,uom:'EA',item_type:'MATERIAL'}]);
const removeNewLine = (i: number) => setNewLineItems(p => p.filter((_,idx)=>idx!==i));
const updateNewLine = (i: number, field: string, val: any) => 
  setNewLineItems(p => p.map((x,idx)=>idx===i?{...x,[field]:val}:x));
const saveNewLines = async () => {
  const valid = newLineItems.filter(l => l.description.trim())
  if (!valid.length) return
  setSavingLines(true)
  try {
    const { data: ex } = await supabase
      .from('job_line_items')
      .select('sort_order')
      .eq('job_id', job.id)
      .order('sort_order', {ascending:false})
      .limit(1)
    const next = ((ex?.[0]?.sort_order) || 0) + 1
    await supabase.from('job_line_items').insert(
      valid.map((x, idx) => ({
        job_id:        job.id,
        description:   x.description.trim(),
        quantity:      x.quantity,
        uom:           x.uom,
        item_type:     x.item_type,
        cost_price:    0,
        sell_price:    0,
        line_total:    0,
        status:        'PENDING',
        sort_order:    next + idx,
        can_spawn_job: true,
      }))
    )
    const { data: r } = await supabase
      .from('job_line_items')
      .select('*, child_job:jobs!child_job_id(job_number)')
      .eq('job_id', job.id)
      .order('sort_order')
    if (r) setLineItems(r.map((li: any) => ({
      ...li,
      child_job_number: li.child_job?.job_number || null
    })))
    setNewLineItems([])
    showMsg('Line items saved')
  } catch(err: any) { showMsg('Error: ' + err.message) }
  finally { setSavingLines(false) }
}
`;

if (content.includes(attachmentsMarker)) {
  content = content.replace(
    new RegExp(attachmentsMarker + '[^;]+;'),
    match => match + stateToInsert
  );
  console.log('✅ State + handlers inserted after attachments');
} else {
  console.warn('⚠️ Could not find attachments state — inserting near top of JobDetailPanel');
  content = content.replace(
    /function JobDetailPanel\([^)]+\)\s*\{/,
    match => match + stateToInsert
  );
}

// 4. FIX RFQ → Job line item copy (uom mapping)
content = content.replace(
  /uom:\s*item\.unit_of_measure/g,
  'uom: item.unit_of_measure || \'EA\''
);
console.log('✅ RFQ line item uom mapping fixed');

    </div>

  {lineItems.length > 0 && (

      </div>
      <table className="w-full text-xs">
        <thead className="bg-gray-50"><tr>
          <th className="px-2 py-1.5 text-left text-gray-500 font-medium w-24">Type</th>

  {lineItems.length === 0 && newLineItems.length === 0 && (

    /import .*lucide-react["']/,
